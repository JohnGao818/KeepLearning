
#### 构建单链表
```c++
#include <stdio.h>
#include <stdlib.h>
#include <cstring>
#include <cmath>

// 单链表
typedef struct LNode
{
    int data;
    struct LNode *next;
}LNode, *LinkList;

void InitList(LinkList &L)
{
    L = (LinkList)malloc(sizeof(LNode));
    L->next = NULL;
}

int Length(LinkList L)
{
    int i = 0;
    LNode *p = L;
    while (p->next != NULL)
    {
        i++;
        p = p->next;
    }
    return i;
}

LNode *LocateElem(LinkList L, int e)
{
    LNode *p = L->next;
    while (p != NULL && p->data != e)
    {
        p = p->next;
    }
    return p;
}

LNode *GetElem(LinkList L, int i)
{
    if (i < 1) {
        return NULL;
    }
    int j = 1;
    LNode *p = L->next;
    while (p != NULL && j < i)
    {
        j++;
        p = p->next;
    }
    return p;
}

bool ListInsert(LinkList &L, int i, int e)
{
    if (i < 1) {
        return false;
    }
    int j = 0;
    LNode *p = L;
    while (p->next != NULL && j < i - 1)
    {
        j++;
        p = p->next;
    }
    if (p->next == NULL)
        return false;
    LNode *s = (LNode*)malloc(sizeof(LNode));
    s->data = e;
    s->next = p->next;
    p->next = s;
    return true;
}

bool ListDelete(LinkList &L, int i, int &e)
{
    if (i < 1) {
        return false;
    }
    int j = 0;
    LNode *p = L, *q;
    while (p->next != NULL && j < i - 1)
    {
        j++;
        p = p->next;
    }
    if (p->next == NULL || j > i-1)
        return false;
    q = p->next;
    p->next = q->next;
    e = q->data;
    free(q);
    return true;
}

void PrintList(LinkList L)
{
    LNode *p = L->next;
    while (p != NULL)
    {
        printf("%d, ", p->data);
        p = p->next;
    }
    
    printf("\n");
}

bool Empty(LinkList L)
{
    return L->next == NULL;
}

LinkList List_TailInsert(LinkList &L) {
    int x;
    scanf("%d", &x);
    LNode *p = L;
    while (x != 9999)
    {
        LNode *s = (LNode*)malloc(sizeof(LNode));
        s->data = x;
        p->next = s;
        p = s;
        scanf("%d", &x);
    }
    p->next = NULL;
    return L;
}

LinkList List_HeaderInsert(LinkList &L) {
    int x;
    scanf("%d", &x);
    while (x != 9999)
    {
        LNode *s = (LNode*)malloc(sizeof(LNode));
        s->data = x;
        s->next = L->next;
        L->next = s;
        scanf("%d", &x);
    }
    return L;
}
```
